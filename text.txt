  <script>
    function calculatePayout() {
      const point = parseInt(document.getElementById("point").value);
      const bet = parseFloat(document.getElementById("bet").value);
      let odds = 0;

      if ([2, 12].includes(point)) odds = 6;
      else if ([3, 11].includes(point)) odds = 3;
      else if ([4, 10].includes(point)) odds = 2;
      else if ([5, 9].includes(point)) odds = 1.5;
      else if ([6, 8].includes(point)) odds = 1.2;
      else odds = 0;

      const oddsWin = bet * odds;
      const passLineWin = bet;
      const totalReturn = bet + oddsWin + passLineWin;

      document.getElementById("result").innerText =
        `You win $${oddsWin.toFixed(2)} on odds and $${passLineWin.toFixed(2)} on the line.\nTotal payout: $${totalReturn.toFixed(2)}`;
    }
  </script>
